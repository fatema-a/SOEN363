SELECT M.title
from movies M
where M.mid IN ((SELECT M1.mid
               FROM movies M1
               WHERE M1.mid IN (SELECT T1.mid
                                FROM tags T1
                                WHERE T1.tid IN (SELECT TN1.tid
                                                 FROM tag_names TN1
                                                 WHERE TN1.tag LIKE '%good%')))
               INTERSECT
               (SELECT M2.mid
               FROM movies M2
               WHERE M2.mid IN (SELECT T2.mid
                                FROM tags T2
                                WHERE T2.tid IN (SELECT TN2.tid
                                                 FROM tag_names TN2
                                                 WHERE TN2.tag LIKE '%bad%'))))
